#!/usr/sbin/nft -f

flush ruleset

table inet filter {
    # Filter traffic destined for the router itself:
    chain input {
        type filter hook input priority 0; policy drop;
        # Allow existing established and related connections
        ct state established,related accept

        # Allow localhost loopback traffic
        iif "lo" accept

        {% if enable_ipv4 %}
        {% if icmp_accept_lan != "" %}
        # Allow specific ICMP queries on lan interface (IPv4)
        iif "{{ interface_lan }}" ip saddr {{ subnet_lan_ipv4 }} icmp type { {{ icmp_accept_lan }} } accept
        {% endif %}

        {% if icmp_accept_wan != "" %}
        # Allow specific ICMP queries on wan interface (IPv4)
        iif "{{ interface_wan }}" icmp type { {{ icmp_accept_wan }} } accept
        {% endif %}
        {% endif %}

        {% if enable_ipv6 %}
        {% if icmpv6_accept_lan != "" %}
        # Allow specific ICMPv6 queries on lan interface
        iif "{{ interface_lan }}" ip6 saddr {{ subnet_lan_ipv6 }} icmpv6 type { {{ icmpv6_accept_lan }} } accept
        {% endif %}

        {% if icmpv6_accept_wan != "" %}
        # Allow specific ICMPv6 queries on wan interface
        iif "{{ interface_wan }}" icmpv6 type { {{ icmpv6_accept_wan }} } accept
        {% endif %}
        {% endif %}

        {% if tcp_accept_lan != "" %}
        # Allow access to specific TCP ports from the lan interface
        {% if enable_ipv4 %}
        ip saddr {{ subnet_lan_ipv4 }} iif "{{ interface_lan }}" tcp dport { {{ tcp_accept_lan }} } accept
        {% endif %}
        {% if enable_ipv6 %}
        ip6 saddr {{ subnet_lan_ipv6 }} iif "{{ interface_lan }}" tcp dport { {{ tcp_accept_lan }} } accept
        {% endif %}
        {% endif %}
        {% if udp_accept_lan != "" %}
        # Allow access to specific UDP ports from the lan interface
        # No source address filter — DHCP clients use 0.0.0.0 before they have a lease
        iif "{{ interface_lan }}" udp dport { {{ udp_accept_lan }} } accept
        {% endif %}

        {% if tcp_accept_wan != "" %}
        # Allow access to specific TCP ports from the wan interface
        iif "{{ interface_wan }}" tcp dport { {{ tcp_accept_wan }} } accept
        {% endif %}

        {% if udp_accept_wan != "" %}
        # Allow access to specific UDP ports from the wan interface
        iif "{{ interface_wan }}" udp dport { {{ udp_accept_wan }} } accept
        {% endif %}

        # Drop and log everything else
        log prefix "Dropped input packet: " drop
    }

    # Filter forwarded traffic (traffic passing through the router):
    # This filters traffic after the NAT prerouting chain modifies the destination route.
    # This filters traffic before the postrouting chain.
    chain forward {
        type filter hook forward priority 0; policy drop;
        # Allow existing established and related connections
        ct state established,related accept

        {% if enable_ipv4 %}
        # Allow IPv4 egress from lan to wan (filtered by allowed destinations)
        ip saddr {{ subnet_lan_ipv4 }} ip daddr { {{ lan_egress_allowed_ipv4 }} } iif "{{ interface_lan }}" oif "{{ interface_wan }}" accept
        {% endif %}

        {% if enable_ipv6 %}
        # Allow IPv6 egress from lan to wan (filtered by allowed destinations)
        ip6 saddr {{ subnet_lan_ipv6 }} ip6 daddr { {{ lan_egress_allowed_ipv6 }} } iif "{{ interface_lan }}" oif "{{ interface_wan }}" accept
        {% endif %}

        {% if tcp_forward_lan.len() > 0 %}
        # DNAT forward TCP from lan
            {% for route in tcp_forward_lan.routes %}
            {% if route.is_ipv4() %}
        ip saddr {{ subnet_lan_ipv4 }} iif "{{ interface_lan }}" ip daddr {{ route.destination_ip }} tcp dport {{ route.destination_port }} accept
            {% else %}
        ip6 saddr {{ subnet_lan_ipv6 }} iif "{{ interface_lan }}" ip6 daddr {{ route.destination_ip }} tcp dport {{ route.destination_port }} accept
            {% endif %}
            {% endfor %}
        {% endif %}
        {% if udp_forward_lan.len() > 0 %}
        # DNAT forward UDP from lan
            {% for route in udp_forward_lan.routes %}
            {% if route.is_ipv4() %}
        ip saddr {{ subnet_lan_ipv4 }} iif "{{ interface_lan }}" ip daddr {{ route.destination_ip }} udp dport {{ route.destination_port }} accept
            {% else %}
        ip6 saddr {{ subnet_lan_ipv6 }} iif "{{ interface_lan }}" ip6 daddr {{ route.destination_ip }} udp dport {{ route.destination_port }} accept
            {% endif %}
            {% endfor %}
        {% endif %}

        {% if tcp_forward_wan.len() > 0 %}
        # DNAT forward TCP from wan
            {% for route in tcp_forward_wan.routes %}
            {% if route.is_ipv4() %}
        iif "{{ interface_wan }}" ip daddr {{ route.destination_ip }} tcp dport {{ route.destination_port }} accept
            {% else %}
        iif "{{ interface_wan }}" ip6 daddr {{ route.destination_ip }} tcp dport {{ route.destination_port }} accept
            {% endif %}
            {% endfor %}
        {% endif %}
        {% if udp_forward_wan.len() > 0 %}
        # DNAT forward UDP from wan
            {% for route in udp_forward_wan.routes %}
            {% if route.is_ipv4() %}
        iif "{{ interface_wan }}" ip daddr {{ route.destination_ip }} udp dport {{ route.destination_port }} accept
            {% else %}
        iif "{{ interface_wan }}" ip6 daddr {{ route.destination_ip }} udp dport {{ route.destination_port }} accept
            {% endif %}
            {% endfor %}
        {% endif %}

        # Log and drop all other packets
        log prefix "Dropped forward packet: " drop
    }

    # Filter traffic whose source is the router itself:
    chain output {
        type filter hook output priority 0; policy drop;
        # Allow all loopback traffic
        oif "lo" accept

        # Allow all outgoing wan traffic
        oif "{{ interface_wan }}" accept

        # Allow all outgoing lan traffic (DHCP responses, DNS, RA, etc.)
        oif "{{ interface_lan }}" accept
    }
}

table inet nat {
    # DNAT for incoming traffic:
    # Modify destination address before routing decision in forwarding chain:
    chain prerouting {
        type nat hook prerouting priority 0; policy accept;
        {% if tcp_forward_lan.len() > 0 %}
        # DNAT from lan
            {% for route in tcp_forward_lan.routes %}
            {% if route.is_ipv4() %}
        iif "{{ interface_lan }}" ip saddr {{ subnet_lan_ipv4 }} tcp dport {{ route.incoming_port }} dnat to {{ route.destination_ip }}:{{ route.destination_port }}
            {% else %}
        iif "{{ interface_lan }}" ip6 saddr {{ subnet_lan_ipv6 }} tcp dport {{ route.incoming_port }} dnat to [{{ route.destination_ip }}]:{{ route.destination_port }}
            {% endif %}
            {% endfor %}
        {% endif %}

        {% if tcp_forward_wan.len() > 0 %}
        # DNAT from wan (TCP)
            {% for route in tcp_forward_wan.routes %}
            {% if route.is_ipv4() %}
        iif "{{ interface_wan }}" tcp dport {{ route.incoming_port }} dnat to {{ route.destination_ip }}:{{ route.destination_port }}
            {% else %}
        iif "{{ interface_wan }}" tcp dport {{ route.incoming_port }} dnat to [{{ route.destination_ip }}]:{{ route.destination_port }}
            {% endif %}
            {% endfor %}
        {% endif %}

        {% if udp_forward_lan.len() > 0 %}
        # DNAT from lan (UDP)
            {% for route in udp_forward_lan.routes %}
            {% if route.is_ipv4() %}
        iif "{{ interface_lan }}" ip saddr {{ subnet_lan_ipv4 }} udp dport {{ route.incoming_port }} dnat to {{ route.destination_ip }}:{{ route.destination_port }}
            {% else %}
        iif "{{ interface_lan }}" ip6 saddr {{ subnet_lan_ipv6 }} udp dport {{ route.incoming_port }} dnat to [{{ route.destination_ip }}]:{{ route.destination_port }}
            {% endif %}
            {% endfor %}
        {% endif %}

        {% if udp_forward_wan.len() > 0 %}
        # DNAT from wan (UDP)
            {% for route in udp_forward_wan.routes %}
            {% if route.is_ipv4() %}
        iif "{{ interface_wan }}" udp dport {{ route.incoming_port }} dnat to {{ route.destination_ip }}:{{ route.destination_port }}
            {% else %}
        iif "{{ interface_wan }}" udp dport {{ route.incoming_port }} dnat to [{{ route.destination_ip }}]:{{ route.destination_port }}
            {% endif %}
            {% endfor %}
        {% endif %}
    }

    # SNAT for outgoing traffic:
    # Modify source address after the forwarding chain, before leaving the router:
    chain postrouting {
        type nat hook postrouting priority 100; policy accept;
        {% if enable_ipv4 %}
        # Masquerade IPv4 LAN-to-WAN traffic (NAT)
        meta nfproto ipv4 oif "{{ interface_wan }}" masquerade
        {% endif %}
        # IPv6 does not use NAT — LAN clients have globally routable addresses.
        # The forward chain's default-deny policy protects LAN from unsolicited inbound.
    }
}
