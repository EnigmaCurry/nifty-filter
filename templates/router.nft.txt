#!/usr/sbin/nft -f

flush ruleset

table ip filter {
    chain input {
        type filter hook input priority 0; policy drop;
        # Allow established and related connections
        ct state established,related accept

        {% if icmp_accept_lan != "" %}
        # Allow specific ICMP queries on lan interface
        ip protocol icmp icmp type { {{ icmp_accept_lan }} } iif "{{ interface_lan }}" accept
        {% endif %}
        {% if icmp_accept_wan != "" %}
        # Allow specific ICMP queries on wan interface
        ip protocol icmp icmp type { {{ icmp_accept_wan }} } iif "{{ interface_wan }}" accept
        {% endif %}
    }

    chain forward {
        type filter hook forward priority 0; policy drop;
        # Allow established and related connections
        ct state established,related accept
        # Allow traffic from local network to internet
        ip saddr {{ subnet_lan }} accept
        # Log and drop all other packets
        log prefix "Dropped forward packet: " drop
    }

    chain output {
        type filter hook output priority 0; policy drop;
    }
}

table ip nat {
    # DNAT for incoming traffic
    chain prerouting {
        type nat hook prerouting priority 0; policy accept;
        # Example for port forwarding (change as needed)
        # tcp dport 80 dnat to 192.168.1.100:80
    }

    # SNAT for outgoing traffic
    chain postrouting {
        type nat hook postrouting priority 100; policy accept;
        oifname "{{ interface_wan }}" iifname "{{ interface_lan }}" masquerade
    }
}

sysctl -w net.ipv4.ip_forward=1
